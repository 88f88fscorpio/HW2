#include "Date.h"
Date::Date()
{
	month = 1;
	day = 1;
	year = 1900;
}
Date::Date(int month, int day, int year)
{
	this->month = month;
	this->day = day;
	this->year = year;
}

void Date::setDate(int month, int day, int year)
{
		this->month = month;
		this->day = day;
		this->year = year;
}

int Date::getYear()
{
	return year;
}

void Date::print()
{
	cout << month << "/" << day << "/" << year;
}
